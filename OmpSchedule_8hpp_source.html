26  //! Representation of OpenMP schedule information: kind and chunk size. This class can be used regardless of
27  //! whether OpenMP is enabled.
28  struct Schedule
29  {
30  //! Special schedule setting which does not change the internal control variables of OpenMP
31  constexpr static std::uint32_t NoSchedule = 0u;
32  //! Integers corresponding to the mandatory OpenMP omp_sched_t enum values as of version 5.0.
33  constexpr static std::uint32_t Static = 1u;
34  constexpr static std::uint32_t Dynamic = 2u;
35  constexpr static std::uint32_t Guided = 3u;
36  constexpr static std::uint32_t Auto = 4u;
37  //! Each schedule value can be combined with monotonic using + or |.
38  constexpr static std::uint32_t Monotonic = 0x80000000u;
39 
40  //! Schedule kind.
41  //!
42  //! We cannot simply use type omp_sched_t since this struct is agnostic of OpenMP. We also cannot create an
43  //! own mirror enum, since OpenMP implementations are allowed to extend the range of values beyond the
44  //! standard ones defined above. So we have to accept and store any uint32_t value, and for non-standard
45  //! values a user has to ensure the underlying implementation supports it.
46  std::uint32_t kind;
47 
48  //! Chunk size. Same as in OpenMP, value 0 corresponds to default chunk size. Using int and not a
49  //! fixed-width type to match OpenMP API.
50  int chunkSize;
51 
52  //! The provided value myKind has to be supported by the underlying OpenMP implementation.
53  //! It does not have to be one of the constants defined above.
54  //! A default-constructed schedule does not affect internal control variables of OpenMP.
55  //! The constructor is constexpr to simplify creation of static constexpr ompSchedule in user code.
56  ALPAKA_FN_HOST constexpr Schedule(std::uint32_t myKind = NoSchedule, int myChunkSize = 0)
57  : kind(myKind)
58  , chunkSize(myChunkSize)
59  {
60  }
61  };
62 
63  //-----------------------------------------------------------------------------
64  //! Get the OpenMP schedule that is applied when the runtime schedule is used.
65  //!
66  //! For OpenMP >= 3.0 returns the value of the internal control variable run-sched-var.
67  //! Without OpenMP or with OpenMP < 3.0, returns the default schedule.
68  //!
69  //! \return Schedule object.
70  ALPAKA_FN_HOST inline auto getSchedule()
71  {
72  // Getting a runtime schedule requires OpenMP 3.0 or newer
73 #if defined _OPENMP && _OPENMP >= 200805
74  omp_sched_t ompKind;
75  int chunkSize = 0;
76  omp_get_schedule(&ompKind, &chunkSize);
77  return Schedule{static_cast<std::uint32_t>(ompKind), chunkSize};
78 #else
79  return Schedule{};
80 #endif
81  }
82 
83  //-----------------------------------------------------------------------------
84  //! Set the OpenMP schedule that is applied when the runtime schedule is used for future parallel regions.
85  //!
86  //! For OpenMP >= 3.0 sets the value of the internal control variable run-sched-var according to the given
87  //! schedule. Without OpenMP or with OpenMP < 3.0, does nothing.
88  //!
89  //! Note that calling from inside a parallel region does not have an immediate effect.
90  ALPAKA_FN_HOST inline void setSchedule(Schedule schedule)
91  {
92  if(schedule.kind != Schedule::NoSchedule)
93  {
94 #if defined _OPENMP && _OPENMP >= 200805
95  omp_set_schedule(static_cast<omp_sched_t>(schedule.kind), schedule.chunkSize);
96 #endif
97  }
98  }
99 
100  } // namespace omp
101 } // namespace alpaka
