41  //! The Omp5 memory copy trait.
42  template<
43  typename TDim,
44  typename TViewDst,
45  typename TViewSrc,
46  typename TExtent>
47  struct TaskCopyOmp5
48  {
49  static_assert(
50  !std::is_const<TViewDst>::value,
51  "The destination view can not be const!");
52 
53  static_assert(
54  Dim<TViewSrc>::value == TDim::value,
55  "The source view is required to have dimensionality TDim!");
56  static_assert(
57  Dim<TViewDst>::value == Dim<TViewSrc>::value,
58  "The source and the destination view are required to have the same dimensionality!");
59  static_assert(
60  Dim<TViewDst>::value == Dim<TExtent>::value,
61  "The views and the extent are required to have the same dimensionality!");
62  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
63  static_assert(
64  std::is_same<Elem<TViewDst>, typename std::remove_const<Elem<TViewSrc>>::type>::value,
65  "The source and the destination view are required to have the same element type!");
66 
67  using Idx = alpaka::Idx<TExtent>;
68 
69  //-----------------------------------------------------------------------------
70  ALPAKA_FN_HOST TaskCopyOmp5(
71  TViewDst & viewDst,
72  TViewSrc const & viewSrc,
73  TExtent const & extent,
74  int const & iDstDevice,
75  int const & iSrcDevice) :
76  m_iDstDevice(iDstDevice),
77  m_iSrcDevice(iSrcDevice),
78  m_extent(castVec<size_t>(extent::getExtentVec(extent))),
79  m_dstPitchBytes(castVec<size_t>(view::getPitchBytesVec(viewDst))),
80  m_srcPitchBytes(castVec<size_t>(view::getPitchBytesVec(viewSrc))),
81  m_dstMemNative(reinterpret_cast<void *>(view::getPtrNative(viewDst))),
82  m_srcMemNative(reinterpret_cast<void const *>(view::getPtrNative(viewSrc)))
83  {
84 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
85  const auto dstExtent(castVec<size_t>(extent::getExtentVec(viewDst)));
86  const auto srcExtent(castVec<size_t>(extent::getExtentVec(viewSrc)));
87  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
88  {
89  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
90  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
91  }
92  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent="
93  << srcExtent << ", dstExtent=" << dstExtent << std::endl;
94 #endif
95  }
96 
97 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
98  //-----------------------------------------------------------------------------
99  ALPAKA_FN_HOST auto printDebug() const
100  -> void
101  {
102  std::cout << __func__
103  << " ddev: " << m_iDstDevice
104  << " ew: " << m_extent
105  << " dptr: " << m_dstMemNative
106  << " sdev: " << m_iSrcDevice
107  << " sptr: " << m_srcMemNative
108  << std::endl;
109  }
110 #endif
111  int m_iDstDevice;
112  int m_iSrcDevice;
113  Vec<TDim, size_t> m_extent;
114  Vec<TDim, size_t> m_dstPitchBytes;
115  Vec<TDim, size_t> m_srcPitchBytes;
116  void * m_dstMemNative;
117  void const * m_srcMemNative;
118 
119  //-----------------------------------------------------------------------------
120  //! Executes the kernel function object.
121  ALPAKA_FN_HOST auto operator()() const
122  -> void
123  {
124  ALPAKA_DEBUG_FULL_LOG_SCOPE;
125 
126 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
127  printDebug();
128 #endif
129  constexpr auto lastDim = TDim::value - 1;
130 
131  if(m_extent.prod() > 0)
132  {
133  // offsets == 0 by ptr shift (?)
134  auto dstOffset(Vec<TDim, size_t>::zeros());
135  auto srcOffset(Vec<TDim, size_t>::zeros());
136 
137  auto dstExtentFull(Vec<TDim, size_t>::zeros());
138  auto srcExtentFull(Vec<TDim, size_t>::zeros());
139 
140  const size_t elementSize =
141  ( m_dstPitchBytes[0]%sizeof(Elem<TViewDst>) || m_srcPitchBytes[0]%sizeof(Elem<TViewDst>) )
142  ? 1 : sizeof(Elem<TViewDst>);
143 
144  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim]/elementSize;
145  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim]/elementSize;
146  for(int i = lastDim - 1; i >= 0; --i)
147  {
148  dstExtentFull[i] = m_dstPitchBytes[i]/m_dstPitchBytes[i+1];
149  srcExtentFull[i] = m_srcPitchBytes[i]/m_srcPitchBytes[i+1];
150  }
151 
152  ALPAKA_OMP5_CHECK(
153  omp_target_memcpy_rect(
154  m_dstMemNative, const_cast<void*>(m_srcMemNative),
155  sizeof(Elem<TViewDst>),
156  TDim::value,
157  reinterpret_cast<size_t const *>(&m_extent),
158  reinterpret_cast<size_t const *>(&dstOffset),
159  reinterpret_cast<size_t const *>(&srcOffset),
160  reinterpret_cast<size_t const *>(&dstExtentFull),
161  reinterpret_cast<size_t const *>(&srcExtentFull),
162  m_iDstDevice, m_iSrcDevice));
163  }
164  }
165  };

201  TViewDst & viewDst,

203  TExtent const & extent,
204  int const & iDstDevice,
205  int const & iSrcDevice) :
206  m_iDstDevice(iDstDevice),

212 #endif

216  {
217 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
218  ALPAKA_ASSERT(m_extentWidth <= m_dstWidth);
219  ALPAKA_ASSERT(m_extentWidth <= m_srcWidth);
220 #endif
221  }

236  << " sptr: " << m_srcMemNative
237  << std::endl;
238  }
239 #endif

250 
251  //-----------------------------------------------------------------------------
252  //! Executes the kernel function object.
253  ALPAKA_FN_HOST auto operator()() const
254  -> void
255  {
256  ALPAKA_DEBUG_FULL_LOG_SCOPE;
257 
258 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL

263  return;

265 
266  ALPAKA_OMP5_CHECK(
