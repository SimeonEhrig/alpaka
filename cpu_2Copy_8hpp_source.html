33  //! The CPU device memory copy task base.
34  //!
35  //! Copies from CPU memory into CPU memory.
36  template<
37  typename TDim,
38  typename TViewDst,
39  typename TViewSrc,
40  typename TExtent>
41  struct TaskCopyCpuBase
42  {
43  using ExtentSize = Idx<TExtent>;
44  using DstSize = Idx<TViewDst>;
45  using SrcSize = Idx<TViewSrc>;
46  using Elem = alpaka::Elem<TViewSrc>;
47 
48  static_assert(
49  !std::is_const<TViewDst>::value,
50  "The destination view can not be const!");
51 
52  static_assert(
53  Dim<TViewDst>::value == Dim<TViewSrc>::value,
54  "The source and the destination view are required to have the same dimensionality!");
55  static_assert(
56  Dim<TViewDst>::value == Dim<TExtent>::value,
57  "The views and the extent are required to have the same dimensionality!");
58  static_assert(
59  Dim<TViewDst>::value == TDim::value,
60  "The destination view and the input TDim are required to have the same dimensionality!");
61 
62  static_assert(
63  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
64  "The destination view and the extent are required to have compatible idx type!");
65  static_assert(
66  meta::IsIntegralSuperset<SrcSize, ExtentSize>::value,
67  "The source view and the extent are required to have compatible idx type!");
68 
69  static_assert(
70  std::is_same<alpaka::Elem<TViewDst>, std::remove_const_t<alpaka::Elem<TViewSrc>>>::value,
71  "The source and the destination view are required to have the same element type!");
72

130  //! The CPU device ND memory copy task.
131  template<
132  typename TDim,
133  typename TViewDst,
134  typename TViewSrc,
135  typename TExtent>
136  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
137  {
138  using DimMin1 = DimInt<TDim::value - 1u>;
139  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
140  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
141  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
142 
143  //-----------------------------------------------------------------------------
144  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
145 
146  //-----------------------------------------------------------------------------
147  ALPAKA_FN_HOST auto operator()() const
148  -> void
149  {
150  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
151 
152 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
153  this->printDebug();
154 #endif
155  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
156  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
157  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
158  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
159  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
160 
161  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
162  {
163  meta::ndLoopIncIdx(
164  extentWithoutInnermost,
165  [&](Vec<DimMin1, ExtentSize> const & idx)
166  {
167  std::memcpy(
168  reinterpret_cast<void *>(this->m_dstMemNative + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
169  reinterpret_cast<void const *>(this->m_srcMemNative + (castVec<SrcSize>(idx) * srcPitchBytesWithoutOutmost).foldrAll(std::plus<SrcSize>())),
170  static_cast<std::size_t>(this->m_extentWidthBytes));
171  });
172  }
173  }
174  };
