33  //! The CPU device ND memory set task base.
34  template<
35  typename TDim,
36  typename TView,
37  typename TExtent>
38  struct TaskSetCpuBase
39  {
40  using ExtentSize = Idx<TExtent>;
41  using DstSize = Idx<TView>;
42  using Elem = alpaka::Elem<TView>;
43 
44  static_assert(
45  !std::is_const<TView>::value,
46  "The destination view can not be const!");
47 
48  static_assert(
49  Dim<TView>::value == Dim<TExtent>::value,
50  "The destination view and the extent are required to have the same dimensionality!");
51  static_assert(
52  Dim<TView>::value == TDim::value,
53  "The destination view and the input TDim are required to have the same dimensionality!");
54 
55  static_assert(
56  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
57  "The view and the extent are required to have compatible idx type!");
58 
59  //-----------------------------------------------------------------------------
60  TaskSetCpuBase(
61  TView & view,
62  std::uint8_t const & byte,
63  TExtent const & extent) :
64  m_byte(byte),
65  m_extent(extent::getExtentVec(extent)),
66  m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem))),
67 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
68  m_dstExtent(extent::getExtentVec(view)),
69 #endif
70  m_dstPitchBytes(view::getPitchBytesVec(view)),
71  m_dstMemNative(reinterpret_cast<std::uint8_t *>(view::getPtrNative(view)))
72  {
73  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
74  ALPAKA_ASSERT(m_extentWidthBytes <= m_dstPitchBytes[TDim::value - 1u]);
75  }

90 #endif

103  //! The CPU device ND memory set task.
104  template<
105  typename TDim,
106  typename TView,
107  typename TExtent>
108  struct TaskSetCpu : public TaskSetCpuBase<TDim, TView, TExtent>
109  {
110  using DimMin1 = DimInt<TDim::value - 1u>;
111  using typename TaskSetCpuBase<TDim, TView, TExtent>::ExtentSize;
112  using typename TaskSetCpuBase<TDim, TView, TExtent>::DstSize;
113 
114  //-----------------------------------------------------------------------------
115  using TaskSetCpuBase<TDim, TView, TExtent>::TaskSetCpuBase;
116 
117  //-----------------------------------------------------------------------------
118  ALPAKA_FN_HOST auto operator()() const
119  -> void
120  {
121  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
122 
123 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
124  this->printDebug();
125 #endif
126  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
127  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
128  // [z, y, x] -> [y, x] because the z pitch (the full idx of the buffer) is not required.
129  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
130 
131  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
132  {
133  meta::ndLoopIncIdx(
134  extentWithoutInnermost,
135  [&](Vec<DimMin1, ExtentSize> const & idx)
136  {
137 
138  memset(
139  reinterpret_cast<void *>(this->m_dstMemNative + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
140  this->m_byte,
141  static_cast<std::size_t>(this->m_extentWidthBytes));
142  });
143  }
144  }
145  };
